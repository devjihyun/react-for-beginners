<!DOCTYPE html>
<html>
    <body>
        <div id="root"></div>
    </body>
    <!-- 1. React CDN Import -->
    <script src="https://unpkg.com/react@17.0.2/umd/react.production.min.js"></script>
    <script src="https://unpkg.com/react-dom@17.0.2/umd/react-dom.production.min.js"></script>
    
    <script>
        // 2. root 가져오기
        const root = document.getElementById("root");
 
        // 3. React JS로 element 생성하기
        const span = React.createElement(
            "span", // html 태그 네임과 동일하게 작성
            { id: "cool-span", style: { color: "red" } }, // 프로퍼티 생성, id, class...
            "Hello I am a span!" // 태그의 content(내용)
            ); 
 
        // 4. span을 root 안에 render
        ReactDOM.render(span, root); // render : React element를 HTML로 만들어 배치, '사용자에게 보여준다'
    </script>
</html>

<!DOCTYPE html>
<html>
<body>
    <div id="root"></div>
</body>
<script src="https://unpkg.com/react@17.0.2/umd/react.production.min.js"></script>
<script src="https://unpkg.com/react-dom@17.0.2/umd/react-dom.production.min.js"></script>
 
<script>
    const root = document.getElementById("root");
    const h3 = React.createElement(
        "h3", {
            id: "title",
            onMouseEnter: () => console.log("mouse enter"), // property에 event listener 등록 가능
        },
        "Hello I am a span!"
    );
 
    const btn = React.createElement(
        "button", {
            onClick: () => console.log("I'm clicked"), // property에 event listener 등록 가능
            style: {
                backgroundColor: "tomato",
            }
        },
        "Click me"
    );
    const container = React.createElement("div", null, [h3, btn]);
    ReactDOM.render(container, root);
</script>
</html>
